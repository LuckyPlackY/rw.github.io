import requests from bs4 import BeautifulSoup # URL источника данных source_url = 'https://example.com/source_page' # Получаем HTML-код страницы response = requests.get(source_url) if response.status_code == 200: # Парсим HTML с помощью BeautifulSoup soup = BeautifulSoup(response.text, 'html.parser') # Пример извлечения всех заголовков

headings = soup.find_all('h1') for heading in headings: print(heading.text) # Выводим текст заголовка else: print(f'Ошибка при получении данных: {response.status_code}')
